## 一、01背包问题

[[0-1背包问题]](https://www.bilibili.com/video/BV19Y411j7uY/?p=5&spm_id_from=333.880.my_history.page.click&vd_source=d0816d50473135dcde378ffd49a30aee)

```c++
#define N 1001
int n，V;//种类和背包容量
int w[N],v[i];//重量与价值
int dp[N][N];
for(int i=1;i<=n;++i)
{
    for(int j=1;j<=V;++j)
    {
        dp[i][j]=max(dp[i-1][j],dp[i-1][j-w[i]]+v[i]);
    }
}
dp[n][V];//最大值


//滚动优化
int dp[N];

for(int i=1;i<=n;++i)
{
    for(int j=V;j>0;--j)
    {
        dp[j]=max(dp[j],dp[j-w[i]]+v[i]);
    }
}
dp[V];//最大值
```



## 二、分组背包



```c++
//比01背包多一个提前二分制优化

int s[N];//个数
int k=0;//编号
int wc[N],vc[N];//真实数据进行01背包的数据
for(int i=1;i<=n;++i)
{
    for(int j=1;j<=s[i];j<<=1)
    {
            k++;
        	wc[k]=j*w[i];
            vc[k]=j*w[i];
            s[i]-=j;
    }
    if(s[i]!=0)
    {
        k++;
        wc[k]=s[i]*w[i];
        vc[k]=s[i]*v[i];
    }
    
}

//现在直接写01背包代码即可wc,和vc是真实数据
```

## 四、最长公共子序列(LCS)

```c++
#include <iostream>
#include <string>
using namespace std;

int dp[N][N];

string a[N],b[N];

a=" "+a;
b=" "+b;

for(int i=1；i<=a.size();++i)
{
    dp[i][0]=dp[i-1][0]+1;
}

for(int j=1;j<=b.size();++j)
{
     dp[0][j]=dp[0][j-1]+1;
}


for(int i=1；i<=a.size();++i)
{
    for(int j=1;j<=b.size();++j)
    {
        if(a[i]==b[j])
        {
            dp[i][j]=dp[i-1][j-1]+1;
        }
        else
        {
            dp[i][j]=max(dp[i-1][j],dp[i][j-1]);
        }
    }
}
dp[a.size()][b.size()];

```



## 五、编辑距离

[编辑距离 - 动态规划解法](https://www.bilibili.com/video/BV15h411Z7Qd/?spm_id_from=333.880.my_history.page.click&vd_source=d0816d50473135dcde378ffd49a30aee)

```c++
	string a,b;
	
    a = " " + a;
    b = " " + b;

    int n = a.size(), m = b.size();
    for (int i = 1; i <= n; ++i)
    {
        dp[i][0] = dp[i - 1][0] + 1;
    }

    for (int j = 1; j <= m; ++j)
    {
        dp[0][j] = dp[0][j-1] + 1;
    }


    for (int i =1; i <=n; i++)
    {
        for (int j = 1; j <=m; j++)
        {
            if (a[i] == b[j])
            {
                dp[i][j] = dp[i - 1][j - 1];
            }
            else
            {
                int temp=min(dp[i - 1][j], dp[i][j - 1]);
                dp[i][j] = min(temp, dp[i - 1][j - 1]) + 1;
            }
        }
    }
    cout << dp[n][m];
```
