### 1.Prim算法

```c++
int prim()
{
    int cost[N][N];//图
    int mincost[N];//到每个节点的最小权值
    bool used[N];//每个节点是否访问过

    int n;//节点数目
    //初始化赋值
    for (int i = 1; i <= n; i++)
    {
        mincost[i] = INT_MAX, used[i] = false;
    }
    //初始化
    mincost[1] = 0;
    int res = 0;//权值大小
    while (true)
    {
        int v = -1;//下一个选中的节点的值

        for (int i = 1; i <= n; i++)
        {
            if (!used[i] && (v == -1 || mincost[i] < mincost[v]))
            {
                v = i;
            }
        }
        if (v == -1)
            break;
        used[v] = true;
        res += mincost[v];
        for (int i = 1; i <= n; i++)
        {
            mincost[i] = min(mincost[i], cost[v][i]);
        }
    }
    return res;
}
```



### 2.kruskal算法

```c++
struct edge
{
    int u, v, cost;
}es[N];

int par[N], rk[N];

void unoin(int r, int l)
{
    r = find(r);
    l = find(l);
    if (r == l)
    {
        return;
    }
    if (rk[r] < rk[l])
    {
        par[r] = l;
    }
    else
    {
        par[l] = r;
        if (rk[r] == rk[l])
            rk[r]++;
    }
}

void init(int  n)
{
    for (int i = 0; i < n; i++)
    {
        par[i] = i, rk[i] = 0;
    }
}

int find(int x)
{
    if (par[x] == x)
    {
        return x;
    }
    else
    {
        par[x] = find(par[x]);
        return par[x];
    }
}

bool same(int v, int u)
{
    return find(v) == find(u);
}

bool cmp(const edge& a, const edge& b)
{
    return a.cost < b.cost;
}

int kruskal()
{

    int n;//节点数目
    int res = 0;//权值大小
    sort(es + 1, es + n + 1, cmp);//排序

    for (int i = 1; i <= n; i++)
    {
        if (!same(es[i].u, es[i].v))
        {
            res += es[i].cost;
            unoin(es[i].u, es[i].v);
        }
    }

    return res;
}
```

## 三、最短路径

### 1.Bellman_ford

```c++
onst int maxn = 2333;

struct Edge
{   
    int to, v;
};

vector<Edge>G[maxn];

int b[maxn];
bool bellmanford(int n,int s)
{
    memset(b, INT_MAX, sizeof(int) * maxn);
    b[s] = 0;//起点
    bool flag = false;
    for (int i = 1; i <= n; i++)
    {
        flag = false;
        for (int m = 1; m <= n; m++)
        {
            for (auto p : G[m])
            {
                int to = p.to, v = p.v;
                if (b[to] > b[m] + v)
                {
                    b[to] = b[m] + v;
                    flag = true;
                }
            }
        }
        if (!flag)
            break;
    }
    return flag;
}
```

```c++
struct Edge
{   
    int to, v;
};

vector<Edge>G[maxn];



bool flag[maxn];//记录是否访问
int par[maxn]; //记录每个节点的最优解的上一个节点
int path[maxn];//记录每个节点到0的最短路径

void dijkstra(int n,int s)//s是起点
{
    for (int i = 1; i <= n; i++)
    {
        flag[i] = false;
        par[i] = -1;
        path[i] = INT_MAX;
    }//进行初始化

    flag[s] = true;
    path[s] = 0;
    for (int i = 1; i <= n; i++)
    {
        int min = INT_MAX;
        int temp=0;
        for (int u = 1; u <= n; u++)
        {
            for (auto p : G[u])
            {
                if (flag[u])
                {
                    int v = p.v, to = p.to;
                    if (path[to] > path[u] + v)
                    {
                        path[to] = path[u] + v;
                        par[to] = u;
                    }                   
                }       
            }
        }
        for (int i = 1; i <= n; i++)
        {  
            if (!flag[i]&&path[i]&&min > path[i])
            {
                min = path[i];
                temp = i;
            }
        }
        flag[temp] = true;
    }
}
```

```c++
void Floyd()
{
    int f[maxn][maxn];
    
	for(int v=1;v<=n;++v)
    {
        for(int i=1;i<=n;++i)
        {
            for(int j=1;j<=n;++j)
            {
                f[i][j]=min(f[i][j],f[v][j]+f[i][v]);
            }
        }
    }
}
```
